```tsx
"use client";
import { AudioLibrary, type Track } from "@clxrity/react-audio";

const tracks: Track[] = [
    {
        src: "./path/to/track.mp3",
    },
    // ...
]

export default function Component() {
    return <AudioLibrary tracks={track} />;
```
